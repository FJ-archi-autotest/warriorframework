[docs]    def open_ssh_session(self, system_name, session_name=None):
        """Connects to the ssh port of the the given system and creates a
        paramiko session object for the system

        A session_id is created using the combintation system_name+session_name
        and returned by this keyword to be stored in Warrior framework's data
        repository

        Tags to be used in input datafile to retrieve values for open_ssh_session
        <ip> = IP address of the system
        <username> = login username of the system
        <password> = login password of the system

        :Arguments:
            1. system_name(string) = Name of the system from the input datafile
            2. session_name(string) = name of the session to the system

        :Returns:
            1. status(bool)= True / False
            2. session_id (dict element)= key, value
        """

        wdesc = "Connect to the ssh port of the system and creates a session"
        Utils.testcase_Utils.pSubStep(wdesc)
        output_dict = {}
        credentials = Utils.data_Utils.get_credentials(self.datafile,
                                                       system_name,
                                                       ['ip', 'username', 'password'])
        logfile = Utils.file_Utils.getCustomLogFile(self.filename,
                                                    self.logsdir, 'ssh_%s_' % system_name)
        Utils.testcase_Utils.pNote(system_name)
        Utils.testcase_Utils.pNote(Utils.file_Utils.getDateTime())
        session_id = Utils.data_Utils.get_session_id(system_name, session_name)

        if credentials is False:
            status = credentials
        else:
            session_object = ssh_utils_class.SSHComm(credentials['ip'],
                                                     uid=credentials['username'],
                                                     pid=credentials['password'], logfile=logfile)
            status = session_object.ssh_con()
            if status == True:
                output_dict[session_id] = session_object
                print_info("Connection to ssh port of the given system {0} successful".\
                    format(credentials['ip']))
            else:
                print_error("Connection to ssh port of the given system {0} failed".\
                    format(credentials['ip']))

        Utils.testcase_Utils.report_substep_status(status)
        return status, output_dict



[docs]    def send_ssh_command(self, command, system_name, session_name=None):
        """Sends a command to the paramiko session of the system
        This keyword expects a paramiko session with the combination of
        system_name and session to be available in the Warrior Framework's
        data repository.
        i.e. the session should have been established using open_ssh_session
        method in this file.
        :Arguments:
            1. command(string)      = the command to be sent to the system
            2. system_name(string)  = Name of the system from the input datafile
            3. session_name(string) = name of the session to the system

        :Returns:
            1. status(bool)= True / False
        """

        wdesc = "Send command to the provided system"
        Utils.testcase_Utils.pSubStep(wdesc)
        Utils.testcase_Utils.pNote(system_name)
        Utils.testcase_Utils.pNote(self.datafile)
        session_id = Utils.data_Utils.get_session_id(system_name, session_name)
        session_object = Utils.data_Utils.get_object_from_datarepository(session_id)
        if session_object:
            status, _ = session_object.get_response(command)
            if status == True:
                print_info("Command executed successfully")
            else:
                print_error("Failed to execute the command %s" % command)
        else:
            print_warning("%s-%s is not available for use" % (system_name, session_name))
            status = False

        Utils.testcase_Utils.report_substep_status(status)
        return status

[docs]    def send_ssh_cmds_from_testdata(self, wdesc, system_name, session_name=None,
                                    row_num=None, title=None):
        """UseAsKeyword=NO
         This keyword will not be displayed in katana GUI
         It behaves as master keyword for send_ssh_cmds_by_testdata
         send_ssh_cmds_by_testdata_row and send_ssh_cmds_by_testdata_title

        :Arguments:
            1. wdesc (string) = description of keyword
            2. system_name (string) = name of the system in the input datafile
            3. session_name(string) = name of the session to the string
            4. row_num (string) = row number in string representation
            5. title (string) = title in string representation

        :Returns:
            1. status(bool)
        """

        Utils.testcase_Utils.pSubStep(wdesc)
        Utils.testcase_Utils.pNote(system_name)

        session_id = Utils.data_Utils.get_session_id(system_name, session_name)
        session_object = Utils.data_Utils.get_object_from_datarepository(session_id)
        credentials = Utils.data_Utils.get_credentials(self.datafile, system_name, ['testdata'])
        testdata = credentials['testdata']
        if session_object:
            status = session_object.send_testdata_cmds(testdata,\
                                                       str_rownum=row_num, strtitle=title)
        else:
            print_warning("%s-%s is not available for use" % (system_name, session_name))
            status = False
        Utils.testcase_Utils.report_substep_status(status)
        return status

            ['remotepath', 'localpath', 'remotefile', 'remotehost', 'remoteuser', 'remotepwd'])

[docs]    def create_nested_session(self, system_name, session_name=None):
        """Creates a SSH session to destination system via a proxy
        Tags to be used in input datafile to retrieve values for
        download_remote_file_sftp
        <via_host> = host that would be connecting to remotehost
        <via_username> = remote login username
        <via_password> = remote login password
        :Arguments:
            1. system_name (string) = name of the system in the input datafile
            2. session_name(string) = name of the session to the string

        :Returns:
            1. status(bool)
            2. output_dict
        """
        wdesc = "Open connection to system via proxy"
        Utils.testcase_Utils.pSubStep(wdesc)
        output_dict = {}
        credentials = Utils.data_Utils.get_credentials(\
                        self.datafile, system_name, ['ip', 'username',
                                                     'password', 'via_host',
                                                     'via_username',
                                                     'via_password'])
        logfile = Utils.file_Utils.getCustomLogFile(self.filename,
                                                    self.logsdir,
                                                    'target_ssh_%s_' % system_name)
    
        Utils.testcase_Utils.pNote(system_name)

        if not credentials:
            status = credentials
        else:
            host_object = ssh_utils_class.SSHComm(\
                                               credentials['via_host'],
                                               uid=credentials['via_username'],
                                               pid=credentials['via_password'],
                                               logfile=logfile)
            status = host_object.ssh_con()
            if status == True:
                dest_obj, status = host_object.connect_target_via_host(\
                                               credentials['ip'],
                                               credentials['username'],
                                               credentials['password'],
                                               log=logfile)
        host_session_id = Utils.data_Utils.get_session_id(system_name,
                                                          '_host')
        dest_session_id = Utils.data_Utils.get_session_id(system_name,
                                                          session_name)
        if status == True:
            output_dict[host_session_id] = host_object
            output_dict[dest_session_id] = dest_obj
            print_info("Connection to ssh port of the given system {0} "
                       "successful".format(credentials['ip']))
        else:
            print_error("Connection to ssh port of the given system {0} "
                        "failed".format(credentials['ip']))

        Utils.testcase_Utils.report_substep_status(status)
        return status, output_dict

[docs]    def run_qtp(self, system_name):
        """Launches QTP to execute a test case

        :Arguments:
            1. system_name(string)  = Name of the system from the input datafile

        :Returns:
            1. status(bool)
        """

        WDesc = "Run QTP test as indicated in the testpath on a specific system_name"
        Utils.testcase_Utils.pSubStep(WDesc)
        print_info(system_name)
        Utils.testcase_Utils.pNote(system_name)
        Utils.testcase_Utils.pNote(Utils.file_Utils.getDateTime())
        credentials = Utils.data_Utils.get_credentials(self.datafile,
                                                       system_name,
                                                       ['domain', 'username', 'password',
                                                        'ip', 'port', 'vbspath', 'vbsname',
                                                        'testpath'])
        domain = credentials["domain"]
        password = base64.b64decode(credentials["password"])
        username = credentials["username"]
        ip = credentials["ip"]
        port = credentials["port"]
        vbspath = credentials["vbspath"]
        vbsname = credentials["vbsname"]
        testpath = credentials["testpath"]
        Utils.testcase_Utils.pNote(testpath)
        print_info(port)

        with open(self.logfile, 'a') as log:
            log.write('Starting subprocess to Run QTP\n')
            log.flush()
            command = 'winexe -U '+ domain +'/'+username+'%'+password+ ' //'+ ip +\
            ' "cmd.exe /c cd ' + vbspath +' & cscript'+' ' + vbsname +' '+ testpath+ '"'
            handle = Popen(command, shell=True, stdout=PIPE, stderr=STDOUT)
            log.write('Outputting everything that the subprocess does...\n')
            log.flush()
            while handle.poll() is None:
                read = handle.communicate()
                status = read[0]
            print status

            log.write('Test PATH: ' + testpath + '\n')
            log.write('System Under Test: ' + system_name + '\n')
            log.write('Test Status: ' + status + '\n')
            log.flush()
            handle.stdout.close()
        status = self.convert_status(str(status).strip())
        Utils.testcase_Utils.report_substep_status(status)
        return status

[docs]    def execute_cse_script(self, system_name, cse_logfile_path='default'):
        """Executes CSE's run script (cse's wrapper script) on the cse view.
           1. Connects to the Server IP
           2. Sets the view
           3. Go to the cse run script location
           4. Execute the cse run script on the server
           5. Send the log file from the server to Warrior logsdir
           6. Checks the log for the run result

           :Arguments:
                 1. system_name(string) = name of the system in the input datafile
                 2. cse_logfile_path = full path to the cse logfile, default='default'

           :Returns:
                1. status(bool) = True or False
        """

        WDesc = "Execute a CSE's  script from Warrior framework "
        print_info("\n **** {0}...... \n".format(WDesc))

        logfile = Utils.file_Utils.getCustomLogFile(self.filename,
                                                    self.logsdir, 'ssh_%s_' % system_name)
        credentials = Utils.data_Utils.get_credentials(self.datafile, system_name,
                                                       ['ip', 'ssh_port', 'username',
                                                        'password', 'prompt'])
        execution_details = Utils.data_Utils.get_credentials(self.datafile,
                                                             system_name,
                                                             ['cse_viewname', 'main_script_path',
                                                              'cse_script_path', 'cse_wrapper',
                                                              'timeout'])
        args_string = Utils.data_Utils.get_cse_script_args_string(self.datafile, system_name)

        ip = credentials['ip']
        username = credentials['username']
        password = credentials['password']
        ssh_port = credentials['ssh_port']
        prompt = credentials['prompt']
        main_script_path = execution_details['main_script_path']
        cse_viewname = execution_details['cse_viewname']
        script_path = execution_details['cse_script_path']
        cse_wrapper = execution_details['cse_wrapper']
        tmout = execution_details['timeout']

        Utils.testcase_Utils.pNote('Script path: {0}'.format(script_path))

        if ssh_port is False or ssh_port is None:
            ssh_port = 22
        if prompt is False or prompt is None:
            prompt = username + '@'
        if tmout is False or tmout is None:
            tmout = None
        else:
            tmout = int(tmout)

        if cse_wrapper is False or cse_wrapper is None:
            cse_wrapper = 'YES'
        if cse_logfile_path == 'default':
            cse_logfile_path = main_script_path+'.log'
        cse_logfile_name = cse_logfile_path.split(os.sep)[-1]

        status = False
        login_status, session_dict = self.login_cse_server(ip, username,
                                                           password, logfile,
                                                           ssh_port='22')
        if login_status:
            session_object = session_dict['cse_ssh_object']
            cse_view_status = self.set_cse_view(session_object, cse_viewname)
            if cse_view_status:
                execution_status = self.cse_script_execution(session_object, script_path,
                                                             main_script_path, cse_logfile_path,
                                                             prompt, cse_wrapper,
                                                             tmout, args_string)
                if execution_status:
                    get_logfile = self.get_cse_logfile_to_warrior(session_object, cse_logfile_path,
                                                                  ip, username, password)
                    if get_logfile:
                        result = self.verify_result_from_cse_logfile(cse_logfile_name)
                        status = result
        return status



[docs]    def login_cse_server(self, ip, username, password, logfile, ssh_port='22'):
        """Login to CSE server
        :Arguments:
            1. ip (string) = ip address of the cse_server
            2. ssh_port(string) = ssh port number
            3. username (string)
            4. password (string)
            5. logfile(string) = logifle path to store CSE server's ssh session output

        :Returns:
        1. status = boolean
        2. ssh_obj (dict element)= key = cse_ssh_object
                                   value = a pexpect session object for the cse serer.

        """

        WDesc = "Login to CSE server"
        Utils.testcase_Utils.pSubStep('Login to CSE server')
        msg = print_info('CSE Server IP: {0}'.format(ip))
        Utils.testcase_Utils.pNote(msg)
        Utils.testcase_Utils.pNote(self.datafile)

        output_dict = {}
        ssh_obj = Utils.cli_Utils.connect_ssh(ip, ssh_port, username, password, logfile)

        if isinstance(ssh_obj, pexpect.spawn):
            status = True
            output_dict['cse_ssh_object'] = ssh_obj
            msg = "Login to cse server {0} sucessful".format(ip)
        else:
            status = False
            msg = "Login to cse server {0} failed".format(ip)

        Utils.testcase_Utils.pNote(msg)
        Utils.testcase_Utils.report_substep_status(status)
        return status, output_dict



[docs]    def set_cse_view(self, cse_ssh_object, cse_viewname):
        """Goto the provided CSE view

        Arguments:
            1. cse_ssh_object (pexpect spawn object) = a session object to the cse_Server
            2. cse_view_name (string) = the name of the cse_view

        Returns
            1. status (bool)
        """

        WDesc = "Goto the CSE view"
        Utils.testcase_Utils.pSubStep(WDesc)
        Utils.testcase_Utils.pNote(self.datafile)
        msg = print_info('Set view: %s' % cse_viewname)
        Utils.testcase_Utils.pNote(msg)

        result = cse_ssh_object.expect([pexpect.TIMEOUT, '(?i)error', '(?i)not found'],
                                           timeout=10)
        if result == 0:
            msg = "CSE view set successfully"
            status = True
        else:
            msg = "Setting CSE view failed"
            error = cse_ssh_object.before
            Utils.testcase_Utils.pNote(error)
            status = False

        Utils.testcase_Utils.pNote(msg)
        Utils.testcase_Utils.report_substep_status(status)

        return status



[docs]    def cse_script_execution(self, cse_ssh_object, script_path, main_script_path,
                             cse_logfile_path, prompt, cse_wrapper='YES',
                             tmout=None, args_string=False):
        """Execute CSE's script

        :Arguments:
            1. cse_ssh_object (pexpect spawn object) = a session object to the cse_Server
            2. script_path = path to the cse script
            3. main_script_path = path to the main script
            4. cse_lofile_path = paht to the cse logfile
            5. prompt = cse server prompt
            6. cse_wrapper = indicates whether the script provided in script_path is
                a cse wrapper script or a normal cse script
                        default = YES

        :Returns:
            status = bool
         """

        WDesc = "Execute CSE run script"
        Utils.testcase_Utils.pNote(WDesc)
        Utils.testcase_Utils.pNote(self.datafile)
        Utils.testcase_Utils.pSubStep("Executing CSE script '{0}'".format(script_path))
        print_info('script_path: {0}'.format(script_path))
        print_info('main script_path: {0}'.format(main_script_path))
        print_info('main script_path: {0}'.format(cse_logfile_path))

        cse_executable_path = '/mvfs/fos-6000/build/sanity/bin/cse'

        run_script_dir = os.path.dirname(script_path)

        Utils.cli_Utils.send_command(cse_ssh_object, '.*', prompt, 'cd %s'% run_script_dir)
        if cse_wrapper.upper() == 'YES':
            command = script_path
        else:
            command = cse_executable_path + ' ' +'run'+ ' ' + script_path

        if args_string is not False:
            command = command + ' ' + args_string

        print_info("Executing command: {0}".format(command))
        Utils.testcase_Utils.pNote("Command : {0}".format(command))
        time.sleep(10)
        cse_ssh_object.sendline(command)
        cse_ssh_object.expect('cse')
        result = cse_ssh_object.expect([pexpect.TIMEOUT, prompt], timeout=tmout)

        if result == 1:
            msg = "cse_run_script_execution completed successfully"
            status = True
        else:
            msg = "Timeout expired"
            status = False

        Utils.testcase_Utils.pNote(msg)
        Utils.testcase_Utils.report_substep_status(status)

        return status



[docs]    def get_cse_logfile_to_warrior(self, session_object, cse_logfile_path, ip, username, password):
        """Gets the result file from cse framework to warrior framework"""

        WDesc = "Get the logfile from CSE framework to Warrior framework's logsdir"
        Utils.testcase_Utils.pSubStep(WDesc)

        cse_logfile_name = cse_logfile_path.split('/')[-1]
        WAccesible_path = '~' + os.sep + cse_logfile_name
        msg = print_info('CSE_logfile: {0}'.format(cse_logfile_path))
        Utils.testcase_Utils.pNote(msg)
        Utils.testcase_Utils.pNote("Copy logfile from '{0}'"\
        "to Warrior accessible path {1}".format(cse_logfile_path, WAccesible_path))

        session_object.sendline('cp {0} ~'.format(cse_logfile_path))
        if session_object.expect([pexpect.TIMEOUT, '\?']) == 1:
            session_object.sendline('y')

        status = Utils.file_Utils.scpGetFile(ip, username, password,
                                             WAccesible_path, self.logsdir,
                                             logfile=None)

        if status:
            msg = "Successfully copied logfile from CSE to Warrior"
        else:
            msg = "Copying logfile from CSE to Warrior failed"

        Utils.testcase_Utils.pNote(msg)
        Utils.testcase_Utils.report_substep_status(status)

        return status



[docs]    def verify_result_from_cse_logfile(self, logfilename):
        '''Parses the result log and returns if the test is PASS or FAIL
           :Arguments:
                 1. result_log (string) = name of the result log file

           :Returns:
                 1. result (bool)'
        '''

        WDesc = "Parse CSE logfile and determine execution passed or failed"
        Utils.testcase_Utils.pSubStep(WDesc)
        result = False
        logfile = self.logsdir + os.sep + logfilename

        Utils.testcase_Utils.pNote("Logfile: {0}".format(logfile))

        try:
            with open(logfile) as log:
                log_lines = log.readlines()
                print_debug(log_lines[-1])

                if '0 (PASSED)' in log_lines[-1]:
                    result = True
            return result

        except IOError as err:
            msg = "I/O error({0}): {1}".format(err.errno, err.strerror)
            Utils.testcase_Utils.pNote(msg)
            return result

        finally:
            log.close()

                                            cse_viewname, wrapper_script_path, main_script_path,
                                            network_config, cse_logfile_path='default',
                                            ssh_port=None, prompt=None, logfile=None):
    """Executes a CSE runsanity script with the details provided by
    user in the Command line interface of warrior """

    print_info("Executiung a cse script {0}".format(wrapper_script_path))
    main_script_name = main_script_path.split(os.sep)[-1]
    if prompt == None:
        prompt = username + '@'
    if ssh_port == None:
        ssh_port = '22'

    print_info('Login to CSE server: {0}'.format(ip))
    ssh_obj = Utils.cli_Utils.connect_ssh(ip, ssh_port, username, password, logfile)
    ssh_obj.sendline('ct setview %s'% cse_viewname)
    ssh_obj.expect(prompt)

    wrapper_script_dir = os.path.dirname(wrapper_script_path)

    if cse_logfile_path == 'default':
        cse_logfile_path = main_script_path+'.log'
        cse_logfile_name = main_script_name+'.log'
    else:
        cse_logfile_name = cse_logfile_path.split('/')[-1]

    print_info('Executing cse wrapper script')
    Utils.cli_Utils.send_command(ssh_obj, '.*', prompt, 'cd %s'% wrapper_script_dir)
    command = '{0} -mainscript {1} -network {2}'.format(wrapper_script_path,
                                                        main_script_path, network_config)
    time.sleep(10)
    print_info(command)
    ssh_obj.sendline(command)
    ssh_obj.expect('running', timeout=None)
    ssh_obj.expect(prompt, timeout=None)

    print_info('get cse logfile in warrior')
    ssh_obj.sendline('cp {0} ~'.format(cse_logfile_path))
    WAccesible_path = '~' + os.sep + cse_logfile_name
    cwd = os.getcwd()
    Utils.file_Utils.scpGetFile(ip, username, password, WAccesible_path, cwd, logfile=None)
    logfile = cwd+os.sep+cse_logfile_name
    print_info('logfile path is : %s' % logfile)

    file_object = open(logfile, "r")
    last_line = file_object.readlines()[-1]
    if '0 (PASSED)' in last_line:
        result = True
    else:
        result = True

    print result
